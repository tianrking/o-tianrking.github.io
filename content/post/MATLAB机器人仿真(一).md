# 二维空间姿态描述
```
>> T1 = SE2(1,2,30*pi/180)


T1 = 
    0.8660   -0.5000         1
    0.5000    0.8660         2
         0         0         1
>> axis([0 5 0 5])
>> trplot2(T1,'frame','1','color','b')
>> T2 = SE2(2,1,0)
 

T2 = 
         1         0         2
         0         1         1
         0         0         1
>> hold on
>> trplot2(T2,'frame','2','color','r')
>> T3 = T1 * T2
 

T3 = 
    0.8660   -0.5000     2.232
    0.5000    0.8660     3.866
         0         0         1
>> trplot2(T3,'frame','3','color','g')
>> T4 = T2 * T1
 

T4 = 
    0.8660   -0.5000         3
    0.5000    0.8660         3
         0         0         1
>> trplot2(T4,'frame','4','color','c')
```
